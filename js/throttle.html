                          # js节流

                          2020-04-16 周四

                          学习计划开始第十三天，今天的目标是理解js节流, 顺便把防抖也一起看了。

                          节流和防抖是通过减少高频触发事件的实际事件处理程序的执行来提高事件处理函数(频繁执行DOM操作、资源加载等重行为，导致UI停顿甚至浏览器崩溃)运行性能的手段，并没有实质上减少事件的触发次数。

                          # 函数防抖(debounce)

                          在事件被触发n秒后再执行回调，如果在这n秒内又被触发，则重新计时。

                          应用场景: 适合多次事件一次响应的情况

                          • 给按钮加函数防抖防止表单多次提交
                          • 对于输入框连续输入进行AJAX验证时，用函数防抖能有效减少请求次数
                          • 判断scroll是否滑到底部，滚动事件+函数防抖
                          // 参照underscore的实现
// 也可以加个取消防抖，参考下面的链接。
function debounce(func, wait, immediate) {
  var timer, result;
  return function () {
    // 解决this指向问题
    var context = this
    // 解决event对象或其他参数传递
    var args = arguments
    if (timer) {
      clearTimeout(timer);
    }
    // immediate 希望立刻执行函数，然后等到停止触发 n 秒后，才可以重新触发执行。
    if (immediate) {
      // 如果已经执行过，不再执行
      var callNow = !timeout;
      timeout = setTimeout(function() {
        timeout = null;
      }, wait)
      if (callNow) {
        result = func.apply(context, args)
      }
    } else {
      timer = setTimeout(function () {
        func.apply(context, args)
      }, wait)
    }
    // 立即执行时，返回防抖函数的返回值。
    return result;
  }
}
var fn1 = function () {
  console.log('boom')
}
// setInterval(debounce(fn1, 500), 1000) // 第一次在1500ms后触发，之后每1000ms触发一次
// setInterval(debounce(fn1, 2000), 1000) // 不会触发一次，每一次都清除定时器，重新计时

// 另一种方式 underscore源码
function debouce2 (func, wait, immediate) {
  var timeout, args, context, timestamp, result;

  var later = function() {
    var last = +new Date() - timestamp;

    if (last < wait && last >= 0) {
      timeout = setTimeout(later, wait - last);
    } else {
      timeout = null;
      if (!immediate) {
        result = func.apply(context, args);
        if (!timeout) context = args = null;
      }
    }
  };

  return function() {
    context = this;
    args = arguments;
    timestamp = +new Date();
    var callNow = immediate && !timeout;
    if (!timeout) timeout = setTimeout(later, wait);
    if (callNow) {
      result = func.apply(context, args);
      context = args = null;
    }

    return result;
  };
};

                          # 函数节流(throttle)

                          规定一个单位时间，在这个单位时间内，只能有一次触发事件的回调函数执行，如果在同一个单位时间内某事件被触发多次，只有一次能生效。

                          应用场景: 适合大量事件按时间做平均分配触发

                          • 游戏中的刷新率
                          • DOM元素拖拽
                          • Canvas画笔功能
                            // 参照underscore的实现, 也可以加个取消防抖，参考下面的链接。
  // 使用时间戳
  function throttle(func, wait) {
    var context, args;
    var previous = 0;
    return function() {
      var now = +new Date();
      context = this;
      args = arguments;
      if (now - previous > wait) {
        func.apply(context, args);
        previous = now;
      }
    }
  }

  // 使用定时器
  function throttle2(func, wait) {
    var context, args, timeout;
    return function() {
      context = this;
      args = arguments;
      if (!timeout) {
        timeout = setTimeout(function(){
          timeout = null;
          func.apply(context, args)
        }, wait)
      }
    }
  }
  let fn2 = () => {
    console.log('boom')
  }
  setInterval(throttle(fn2, 1000), 10) // 一秒打印出一次boom

                          时间戳与定时器两种方式比较:

                          • 时间戳会立即执行，定时器会在 n 秒后第一次执行
                          • 时间戳停止触发后没有办法再执行事件, 定时器停止触发后依然会再执行一次事件

                          考虑将二者结合起来？

                            /* 
   * options.leading = false 表示禁用第一次执行
   * options.trailing = false 表示禁用停止触发的回调
   * 注意二者不能同时设置，只能有一个生效。默认首尾都会调用
  */
  function throttle3(func, wait, options) {
    var timeout, context, args, result;
    var previous = 0;
    if (!options) options = {};
    var later = function() {
      previous = options.leading === false ? 0 : new Date().getTime();
      timeout = null;
      func.apply(context, args);
      if (!timeout) context = args = null;
    };
    var throttled = function() {
      var now = new Date().getTime();
      if (!previous && options.leading === false) previous = now;
      // 下次触发 func 剩余的时间
      var remaining = wait - (now - previous);
      context = this;
      args = arguments;
      // 如果没有剩余的时间了或者你改了系统时间
      if (remaining <= 0 || remaining > wait) {
        // 由于setTimeout存在最小时间精度问题，因此会存在到达wait的时间间隔，但之前设置的setTimeout操作还没被执行，因此为保险起见，这里先清理setTimeout操作
        if (timeout) {
          clearTimeout(timeout);
          timeout = null;
        }
        previous = now;
        func.apply(context, args);
        if (!timeout) context = args = null;
      } else if (!timeout && options.trailing !== false) {
        timeout = setTimeout(later, remaining);
      }
    };
    return throttled;
  }

                          节流演示: 通过绑定鼠标移动事件来触发效果， 支持首尾都触发一次.

                          总结: 二者都是在时间轴上控制函数的执行次数, 防抖可以类比为电梯不断上乘客,节流可以看做幻灯片限制频率播放电影

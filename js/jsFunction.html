                          # 安全的类型检查

                          typeof() 以及 instance有不足之处.

                          // Array是window的属性，当存在多个全局作用域(多个frame)时，会返回不同的结果.
var isArray = value instanceof Array

                          原生数组的构造函数名与全局作用域无关，使用toString()就能保证返回一致的值.

                          function isArray(value){
  return Object.prototype.toString.call(value) == "[object Array]";
}
// 也可以基于这一思路来测试某个值是不是原生函数或者正则表达式.
function isRegExp(value){
  return Object.prototype.toString.call(value) == "[object RegExp]";
}
// 检测原生JSON对象
var isNativeJSON = window.JSON && Object.prototype.toString.call(JSON) == "[object JSON]";

                          # 作用域安全的构造函数

                          防止构造函数不通过new的方式直接运行，而导致了this执行了其他对象(window).

                          function Person(name, age, job) {
  if (this instanceof Person) {
    this.name = name;
    this.age = age;
    this.job = job;
  } else {
    return new Person(name, age, job);
  }
}
// new Person() 与 Person()都可以。
// 注意，这样会锁定调用构造函数的环境. 不适用单独的call继承方式。可以配合原型继承就不会有问题.
